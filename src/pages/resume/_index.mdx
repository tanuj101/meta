import Link from '@/components/Link.astro'

import { site } from '@/stores/site'
import { recivi } from '@/stores/recivi'

export const age = new Date().getFullYear() - site.birthYear

export const hobbyYears = age - site.hobbyAge
export const hobbyPercent = Math.ceil((hobbyYears / age) * 100)

export const jobYears = age - site.jobAge
export const jobPercent = Math.ceil((jobYears / hobbyYears) * 100)

export const githubUrl = recivi.bio.profiles?.find(
  (profile) => profile.site.id === 'github' || profile.site.name === 'GitHub'
)?.url
